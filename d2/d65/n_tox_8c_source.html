637 /* basic wrap, ignores embedded '\t', '\n' or '|'
638  * inserts continuation markers if there's enough space left,
639  * otherwise turns spaces into newlines if possible */
640 void wrap(char output[STRING_LENGTH_WRAPPED], char input[STRING_LENGTH], int line_width)
641 {
642  size_t i, len = strlen(input);
643 
644  if ((line_width < 4) || (len < (size_t)line_width)) {
645  /* if line_width ridiculously tiny, it's not worth the effort */
646  strcpy(output, input);
647  return;
648  }
649 
650  /* how much can we shift? */
651  size_t delta_is = 0, delta_remain = STRING_LENGTH_WRAPPED - len - 1;
652 
653  /* if the line is very very short, don't insert continuation markers,
654  * as they would use up too much of the line */
655  if ((size_t)line_width < 2 * wrap_cont_len)
656  delta_remain = 0;
657 
658  for (i = line_width; i < len; i += line_width) {
659  /* look backward for a space to expand/turn into a new line */
660  size_t k = i;
661  size_t m = i - line_width;
662 
663  while (input[k] != ' ' && k > m) {
664  k--;
665  }
666 
667  if (k > m) {
668  if (delta_remain > wrap_cont_len) {
669  /* replace space with continuation, then
670  * set the pos. after the space as new line start
671  * (i.e. space is being "eaten") */
672  memcpy(output + m + delta_is, input + m, k - m);
673  strcpy(output + k + delta_is, wrap_cont_str);
674 
675  delta_remain -= wrap_cont_len - 1;
676  delta_is += wrap_cont_len - 1;
677  i = k + 1;
678  } else {
679  /* no more space to push forward: replace the space,
680  * use its pos. + 1 as starting point for the next line */
681  memcpy(output + m + delta_is, input + m, k - m);
682  output[k + delta_is] = '\n';
683  i = k + 1;
684  }
685  } else {
686  /* string ends right here:
687  * don't add a continuation marker with nothing following */
688  if (i == len - 1)
689  break;
690 
691  /* nothing found backwards */
692  if (delta_remain > wrap_cont_len) {
693  /* break at the end of the line,
694  * i.e. in the middle of the word at the border */
695  memcpy(output + m + delta_is, input + m, line_width);
696  strcpy(output + i + delta_is, wrap_cont_str);
697 
698  delta_remain -= wrap_cont_len;
699  delta_is += wrap_cont_len;
700  } else {
701  /* no more space to push, no space to convert:
702  * just copy the whole line and move on;
703  * means the line count calc'ed will be off */
704  memcpy(output + m + delta_is, input + m, line_width);
705  }
706  }
707  }
708 
709  i -= line_width;
710  memcpy(output + i + delta_is, input + i, len - i);
711 
712  output[len + delta_is] = 0;
713 }
714 
715 /*
716  * extended wrap, honors '\n', accepts '|' as "break here when necessary"
717  * marks wrapped lines with "+ " in front, which does expand output
718  * does NOT honor '\t': would require a lot more work (and tab width isn't always 8)
719  */
720 void wrap_bars(char output[STRING_LENGTH_WRAPPED], char input[STRING_LENGTH], size_t line_width)
721 {
722  size_t len = strlen(input);
723  size_t ipos, opos = 0;
724  size_t bar_avail = 0, space_avail = 0, nl_got = 0; /* in opos */
725 
726  for (ipos = 0; ipos < len; ipos++) {
727  if (opos - nl_got < line_width) {
728  /* not yet at the limit */
729  char c = input[ipos];
730 
731  if (c == ' ')
732  space_avail = opos;
733 
734  output[opos++] = input[ipos];
735 
736  if (opos >= STRING_LENGTH_WRAPPED) {
737  opos = STRING_LENGTH_WRAPPED - 1;
738  break;
739  }
740 
741  if (c == '|') {
742  output[opos - 1] = ' ';
743  bar_avail = opos;
744 
745  if (opos + 2 >= STRING_LENGTH_WRAPPED) {
746  opos = STRING_LENGTH_WRAPPED - 1;
747  break;
748  }
749 
750  output[opos++] = '|';
751  output[opos++] = ' ';
752  }
753 
754  if (c == '\n')
755  nl_got = opos;
756 
757  continue;
758  } else {
759  /* at the limit */
760  if (bar_avail > nl_got) {
761  /* overwrite */
762  memcpy(output + bar_avail - 1, wrap_cont_str, wrap_cont_len);
763  nl_got = bar_avail;
764 
765  ipos--;
766  continue;
767  }
768 
769  if (space_avail > nl_got) {
770  if (opos + wrap_cont_len - 1 >= STRING_LENGTH_WRAPPED) {
771  opos = STRING_LENGTH_WRAPPED - 1;
772  break;
773  }
774 
775  /* move forward by 2 characters */
776  memmove(output + space_avail + 3, output + space_avail + 1, opos - (space_avail + 1));
777  memcpy(output + space_avail, wrap_cont_str, wrap_cont_len);
778  nl_got = space_avail + 1;
779 
780  opos += 2;
781  ipos--;
782  continue;
783  }
784 
785  char c = input[ipos];
786 
787  if ((c == '|') || (c == ' ') || (c == '\n')) {
788  if (opos + wrap_cont_len >= STRING_LENGTH_WRAPPED) {
789  opos = STRING_LENGTH_WRAPPED - 1;
790  break;
791  }
792 
793  memcpy(output + opos, wrap_cont_str, wrap_cont_len);
794 
795  nl_got = opos;
796  opos += wrap_cont_len;
797  }
798 
799  output[opos++] = input[ipos];
800 
801  if (opos >= STRING_LENGTH_WRAPPED) {
802  opos = STRING_LENGTH_WRAPPED - 1;
803  break;
804  }
805 
806  continue;
807  }
808  }
809 
810  if (opos >= STRING_LENGTH_WRAPPED)
811  opos = STRING_LENGTH_WRAPPED - 1;
812 
813  output[opos] = 0;
814 }
