149 /* Add a TCP relay associated to the friend.
150  *
151  * return -1 on failure.
152  * return 0 on success.
153  */
154 int friend_add_tcp_relay(Friend_Connections *fr_c, int friendcon_id, IP_Port ip_port, const uint8_t *public_key)
155 {
156  Friend_Conn *friend_con = get_conn(fr_c, friendcon_id);
157 
158  if (!friend_con)
159  return -1;
160 
161  /* Local ip and same pk means that they are hosting a TCP relay. */
162  if (Local_ip(ip_port.ip) && memcmp(friend_con->dht_temp_pk, public_key, crypto_box_PUBLICKEYBYTES) == 0) {
163  if (friend_con->dht_ip_port.ip.family != 0) {
164  ip_port.ip = friend_con->dht_ip_port.ip;
165  } else {
166  friend_con->hosting_tcp_relay = 0;
167  }
168  }
169 
170  unsigned int i;
171 
172  uint16_t index = friend_con->tcp_relay_counter % FRIEND_MAX_STORED_TCP_RELAYS;
173 
174  for (i = 0; i < FRIEND_MAX_STORED_TCP_RELAYS; ++i) {
175  if (friend_con->tcp_relays[i].ip_port.ip.family != 0
176  && memcmp(friend_con->tcp_relays[i].public_key, public_key, crypto_box_PUBLICKEYBYTES) == 0) {
177  memset(&friend_con->tcp_relays[i], 0, sizeof(Node_format));
178  }
179  }
180 
181  friend_con->tcp_relays[index].ip_port = ip_port;
182  memcpy(friend_con->tcp_relays[index].public_key, public_key, crypto_box_PUBLICKEYBYTES);
183  ++friend_con->tcp_relay_counter;
184 
185  return add_tcp_relay_peer(fr_c->net_crypto, friend_con->crypt_connection_id, ip_port, public_key);
186 }

241  return 1;
242  }
243 
244  return 0;
245 }

309 static int handle_status(void *object, int number, uint8_t status)
310 {
311  Friend_Connections *fr_c = object;
312  Friend_Conn *friend_con = get_conn(fr_c, number);
313 
314  if (!friend_con)
315  return -1;
316 
317  _Bool call_cb = 0;
318 
319  if (status) { /* Went online. */
320  call_cb = 1;
321  friend_con->status = FRIENDCONN_STATUS_CONNECTED;
322  friend_con->ping_lastrecv = unix_time();
323  friend_con->share_relays_lastsent = 0;
324  onion_set_friend_online(fr_c->onion_c, friend_con->onion_friendnum, status);
325  } else { /* Went offline. */
326  if (friend_con->status != FRIENDCONN_STATUS_CONNECTING) {
327  call_cb = 1;
328  friend_con->dht_pk_lastrecv = unix_time();
329  onion_set_friend_online(fr_c->onion_c, friend_con->onion_friendnum, status);
330  }
331 
332  friend_con->status = FRIENDCONN_STATUS_CONNECTING;
333  friend_con->crypt_connection_id = -1;
334  friend_con->hosting_tcp_relay = 0;
335  }
336 
337  if (call_cb) {
338  unsigned int i;
339 
340  for (i = 0; i < MAX_FRIEND_CONNECTION_CALLBACKS; ++i) {
341  if (friend_con->callbacks[i].status_callback)
342  friend_con->callbacks[i].status_callback(friend_con->callbacks[i].status_callback_object,
343  friend_con->callbacks[i].status_callback_id, status);
344  }
345  }
346 
347  return 0;
348 }

375 static int handle_packet(void *object, int number, uint8_t *data, uint16_t length)
376 {
377  if (length == 0)
378  return -1;
379 
380  Friend_Connections *fr_c = object;
381  Friend_Conn *friend_con = get_conn(fr_c, number);
382 
383  if (!friend_con)
384  return -1;
385 
386  if (data[0] == PACKET_ID_FRIEND_REQUESTS) {
387  if (fr_c->fr_request_callback)
388  fr_c->fr_request_callback(fr_c->fr_request_object, friend_con->real_public_key, data, length);
389 
390  return 0;
391  } else if (data[0] == PACKET_ID_ALIVE) {
392  friend_con->ping_lastrecv = unix_time();
393  return 0;
394  } else if (data[0] == PACKET_ID_SHARE_RELAYS) {
395  Node_format nodes[MAX_SHARED_RELAYS];
396  int n;
397 
398  if ((n = unpack_nodes(nodes, MAX_SHARED_RELAYS, NULL, data + 1, length - 1, 1)) == -1)
399  return -1;
400 
401  int j;
402 
403  for (j = 0; j < n; j++) {
404  friend_add_tcp_relay(fr_c, number, nodes[j].ip_port, nodes[j].public_key);
405  }
406 
407  return 0;
408  }
409 
410  unsigned int i;
411 
412  for (i = 0; i < MAX_FRIEND_CONNECTION_CALLBACKS; ++i) {
413  if (friend_con->callbacks[i].data_callback)
414  friend_con->callbacks[i].data_callback(friend_con->callbacks[i].data_callback_object,
415  friend_con->callbacks[i].data_callback_id, data, length);
416 
417  friend_con = get_conn(fr_c, number);
418 
419  if (!friend_con)
420  return -1;
421  }
422 
423  return 0;
424 }

790 /* main friend_connections loop. */
791 void do_friend_connections(Friend_Connections *fr_c)
792 {
793  uint32_t i;
794  uint64_t temp_time = unix_time();
795 
796  for (i = 0; i < fr_c->num_cons; ++i) {
797  Friend_Conn *friend_con = get_conn(fr_c, i);
798 
799  if (friend_con) {
800  if (friend_con->status == FRIENDCONN_STATUS_CONNECTING) {
801  if (friend_con->dht_pk_lastrecv + FRIEND_DHT_TIMEOUT < temp_time) {
802  if (friend_con->dht_lock) {
803  DHT_delfriend(fr_c->dht, friend_con->dht_temp_pk, friend_con->dht_lock);
804  friend_con->dht_lock = 0;
805  }
806  }
807 
808  if (friend_con->dht_ip_port_lastrecv + FRIEND_DHT_TIMEOUT < temp_time) {
809  friend_con->dht_ip_port.ip.family = 0;
810  }
811 
812  if (friend_con->dht_lock) {
813  if (friend_new_connection(fr_c, i) == 0) {
814  set_direct_ip_port(fr_c->net_crypto, friend_con->crypt_connection_id, friend_con->dht_ip_port, 0);
815  connect_to_saved_tcp_relays(fr_c, i, (MAX_FRIEND_TCP_CONNECTIONS / 2)); /* Only fill it half up. */
816  }
817  }
818 
819  } else if (friend_con->status == FRIENDCONN_STATUS_CONNECTED) {
820  if (friend_con->ping_lastsent + FRIEND_PING_INTERVAL < temp_time) {
821  send_ping(fr_c, i);
822  }
823 
824  if (friend_con->share_relays_lastsent + SHARE_RELAYS_INTERVAL < temp_time) {
825  send_relays(fr_c, i);
826  }
827 
828  if (friend_con->ping_lastrecv + FRIEND_CONNECTION_TIMEOUT < temp_time) {
829  /* If we stopped receiving ping packets, kill it. */
830  crypto_kill(fr_c->net_crypto, friend_con->crypt_connection_id);
831  friend_con->crypt_connection_id = -1;
832  handle_status(fr_c, i, 0); /* Going offline. */
833  }
834  }
835  }
836  }
837 
838  LANdiscovery(fr_c);
839 }
