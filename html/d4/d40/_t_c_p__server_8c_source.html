38 static int bind_to_port(sock_t sock, int family, uint16_t port)
39 {
40  struct sockaddr_storage addr = {0};

44  struct sockaddr_in *addr4 = (struct sockaddr_in *)&addr;
45 
46  addrsize = sizeof(struct sockaddr_in);
47  addr4->sin_family = AF_INET;
48  addr4->sin_port = htons(port);

318 /* return 0 if pending data was sent completely
319  * return -1 if it wasn't
320  */
321 static int send_pending_data(TCP_Secure_Connection *con)
322 {
323  /* finish sending current non-priority packet */
324  if (send_pending_data_nonpriority(con) == -1) {
325  return -1;
326  }
327 
328  TCP_Priority_List *p = con->priority_queue_start;
329 
330  while (p) {
331  uint16_t left = p->size - p->sent;
332  int len = send(con->sock, p->data + p->sent, left, MSG_NOSIGNAL);
333 
334  if (len != left) {
335  if (len > 0) {
336  p->sent += len;
337  }
338 
339  break;
340  }
341 
342  TCP_Priority_List *pp = p;
343  p = p->next;
344  free(pp);
345  }
346 
347  con->priority_queue_start = p;
348 
349  if (!p) {
350  con->priority_queue_end = NULL;
351  return 0;
352  }
353 
354  return -1;
355 }

384 /* return 1 on success.
385  * return 0 if could not send packet.
386  * return -1 on failure (connection must be killed).
387  */
388 static int write_packet_TCP_secure_connection(TCP_Secure_Connection *con, const uint8_t *data, uint16_t length,
389  _Bool priority)
390 {
391  if (length + crypto_box_MACBYTES > MAX_PACKET_SIZE)
392  return -1;
393 
394  _Bool sendpriority = 1;
395 
396  if (send_pending_data(con) == -1) {
397  if (priority) {
398  sendpriority = 0;
399  } else {
400  return 0;
401  }
402  }
403 
404  uint8_t packet[sizeof(uint16_t) + length + crypto_box_MACBYTES];
405 
406  uint16_t c_length = htons(length + crypto_box_MACBYTES);
407  memcpy(packet, &c_length, sizeof(uint16_t));
408  int len = encrypt_data_symmetric(con->shared_key, con->sent_nonce, data, length, packet + sizeof(uint16_t));
409 
410  if ((unsigned int)len != (sizeof(packet) - sizeof(uint16_t)))
411  return -1;
412 
413  if (priority) {
414  len = sendpriority ? send(con->sock, packet, sizeof(packet), MSG_NOSIGNAL) : 0;
415 
416  if (len <= 0) {
417  len = 0;
418  }
419 
420  increment_nonce(con->sent_nonce);
421 
422  if ((unsigned int)len == sizeof(packet)) {
423  return 1;
424  }
425 
426  return add_priority(con, packet, sizeof(packet), len);
427  }
428 
429  len = send(con->sock, packet, sizeof(packet), MSG_NOSIGNAL);
430 
431  if (len <= 0)
432  return 0;
433 
434  increment_nonce(con->sent_nonce);
435 
436  if ((unsigned int)len == sizeof(packet))
437  return 1;
438 
439  memcpy(con->last_packet, packet, sizeof(packet));
440  con->last_packet_length = sizeof(packet);
441  con->last_packet_sent = len;
442  return 1;
443 }

480 /* return 1 if everything went well.
481  * return -1 if the connection must be killed.
482  */
483 static int handle_TCP_handshake(TCP_Secure_Connection *con, const uint8_t *data, uint16_t length,
484  const uint8_t *self_secret_key)
485 {
486  if (length != TCP_CLIENT_HANDSHAKE_SIZE)
487  return -1;
488 
489  if (con->status != TCP_STATUS_CONNECTED)
490  return -1;
491 
492  uint8_t shared_key[crypto_box_BEFORENMBYTES];
493  encrypt_precompute(data, self_secret_key, shared_key);
494  uint8_t plain[TCP_HANDSHAKE_PLAIN_SIZE];
495  int len = decrypt_data_symmetric(shared_key, data + crypto_box_PUBLICKEYBYTES,
496  data + crypto_box_PUBLICKEYBYTES + crypto_box_NONCEBYTES, TCP_HANDSHAKE_PLAIN_SIZE + crypto_box_MACBYTES, plain);
497 
498  if (len != TCP_HANDSHAKE_PLAIN_SIZE)
499  return -1;
500 
501  memcpy(con->public_key, data, crypto_box_PUBLICKEYBYTES);
502  uint8_t temp_secret_key[crypto_box_SECRETKEYBYTES];
503  uint8_t resp_plain[TCP_HANDSHAKE_PLAIN_SIZE];
504  crypto_box_keypair(resp_plain, temp_secret_key);
505  random_nonce(con->sent_nonce);
506  memcpy(resp_plain + crypto_box_PUBLICKEYBYTES, con->sent_nonce, crypto_box_NONCEBYTES);
507  memcpy(con->recv_nonce, plain + crypto_box_PUBLICKEYBYTES, crypto_box_NONCEBYTES);
508 
509  uint8_t response[TCP_SERVER_HANDSHAKE_SIZE];
510  new_nonce(response);
511 
512  len = encrypt_data_symmetric(shared_key, response, resp_plain, TCP_HANDSHAKE_PLAIN_SIZE,
513  response + crypto_box_NONCEBYTES);
514 
515  if (len != TCP_HANDSHAKE_PLAIN_SIZE + crypto_box_MACBYTES)
516  return -1;
517 
518  if (TCP_SERVER_HANDSHAKE_SIZE != send(con->sock, response, TCP_SERVER_HANDSHAKE_SIZE, MSG_NOSIGNAL))
519  return -1;
520 
521  encrypt_precompute(plain, temp_secret_key, con->shared_key);
522  con->status = TCP_STATUS_UNCONFIRMED;
523  return 1;
524 }

579 static int handle_TCP_routing_req(TCP_Server *TCP_server, uint32_t con_id, const uint8_t *public_key)
580 {
581  uint32_t i;
582  uint32_t index = ~0;
583  TCP_Secure_Connection *con = &TCP_server->accepted_connection_array[con_id];
584 
585  /* If person tries to cennect to himself we deny the request*/
586  if (memcmp(con->public_key, public_key, crypto_box_PUBLICKEYBYTES) == 0) {
587  if (send_routing_response(con, 0, public_key) == -1)
588  return -1;
589 
590  return 0;
591  }
592 
593  for (i = 0; i < NUM_CLIENT_CONNECTIONS; ++i) {
594  if (con->connections[i].status != 0) {
595  if (memcmp(public_key, con->connections[i].public_key, crypto_box_PUBLICKEYBYTES) == 0) {
596  if (send_routing_response(con, i + NUM_RESERVED_PORTS, public_key) == -1) {
597  return -1;
598  } else {
599  return 0;
600  }
601  }
602  } else if (index == (uint32_t)~0) {
603  index = i;
604  }
605  }
606 
607  if (index == (uint32_t)~0) {
608  if (send_routing_response(con, 0, public_key) == -1)
609  return -1;
610 
611  return 0;
612  }
613 
614  int ret = send_routing_response(con, index + NUM_RESERVED_PORTS, public_key);
615 
616  if (ret == 0)
617  return 0;
618 
619  if (ret == -1)
620  return -1;
621 
622  con->connections[index].status = 1;
623  memcpy(con->connections[index].public_key, public_key, crypto_box_PUBLICKEYBYTES);
624  int other_index = get_TCP_connection_index(TCP_server, public_key);
625 
626  if (other_index != -1) {
627  uint32_t other_id = ~0;
628  TCP_Secure_Connection *other_conn = &TCP_server->accepted_connection_array[other_index];
629 
630  for (i = 0; i < NUM_CLIENT_CONNECTIONS; ++i) {
631  if (other_conn->connections[i].status == 1
632  && memcmp(other_conn->connections[i].public_key, con->public_key, crypto_box_PUBLICKEYBYTES) == 0) {
633  other_id = i;
634  break;
635  }
636  }
637 
638  if (other_id != (uint32_t)~0) {
639  con->connections[index].status = 2;
640  con->connections[index].index = other_index;
641  con->connections[index].other_id = other_id;
642  other_conn->connections[other_id].status = 2;
643  other_conn->connections[other_id].index = con_id;
644  other_conn->connections[other_id].other_id = index;
645  //TODO: return values?
646  send_connect_notification(con, index);
647  send_connect_notification(other_conn, other_id);
648  }
649  }
650 
651  return 0;
652 }

738  * return -1 on failure
739  */
740 static int handle_TCP_packet(TCP_Server *TCP_server, uint32_t con_id, const uint8_t *data, uint16_t length)
741 {

743  return -1;
744 
745  TCP_Secure_Connection *con = &TCP_server->accepted_connection_array[con_id];
746 
747  switch (data[0]) {
748  case TCP_PACKET_ROUTING_REQUEST: {
749  if (length != 1 + crypto_box_PUBLICKEYBYTES)
750  return -1;
751 
752  return handle_TCP_routing_req(TCP_server, con_id, data + 1);
753  }
754 
755  case TCP_PACKET_CONNECTION_NOTIFICATION: {
756  if (length != 2)
757  return -1;
758 
759  break;
760  }
761 
762  case TCP_PACKET_DISCONNECT_NOTIFICATION: {
763  if (length != 2)
764  return -1;
765 
766  return rm_connection_index(TCP_server, con, data[1] - NUM_RESERVED_PORTS);
767  }
768

770  if (length != 1 + sizeof(uint64_t))
771  return -1;
772

774  response[0] = TCP_PACKET_PONG;
775  memcpy(response + 1, data + 1, sizeof(uint64_t));
776  write_packet_TCP_secure_connection(con, response, sizeof(response), 1);
777  return 0;
778  }
779 
780  case TCP_PACKET_PONG: {
781  if (length != 1 + sizeof(uint64_t))
782  return -1;
783

785  memcpy(&ping_id, data + 1, sizeof(uint64_t));
786 
787  if (ping_id) {
788  if (ping_id == con->ping_id) {
789  con->ping_id = 0;
790  }
791 
792  return 0;
793  } else {
794  return -1;
795  }
796  }
797 
798  case TCP_PACKET_OOB_SEND: {
799  if (length <= 1 + crypto_box_PUBLICKEYBYTES)
800  return -1;
801 
802  return handle_TCP_oob_send(TCP_server, con_id, data + 1, data + 1 + crypto_box_PUBLICKEYBYTES,
803  length - (1 + crypto_box_PUBLICKEYBYTES));
804  }
805 
806  case TCP_PACKET_ONION_REQUEST: {
807  if (TCP_server->onion) {
808  if (length <= 1 + crypto_box_NONCEBYTES + ONION_SEND_BASE * 2)
809  return -1;
810

812  source.port = 0; // dummy initialise
813  source.ip.family = TCP_ONION_FAMILY;
814  source.ip.ip6.uint32[0] = con_id;
815  source.ip.ip6.uint32[1] = 0;
816  source.ip.ip6.uint64[1] = con->identifier;
817  onion_send_1(TCP_server->onion, data + 1 + crypto_box_NONCEBYTES, length - (1 + crypto_box_NONCEBYTES), source,
818  data + 1);
819  }
820 
821  return 0;
822  }
823 
824  case TCP_PACKET_ONION_RESPONSE: {
825  return -1;
826  }
827

830  return -1;

835  return -1;
836 
837  if (con->connections[c_id].status == 0)
838  return -1;
839 
840  if (con->connections[c_id].status != 2)
841  return 0;
842 
843  uint32_t index = con->connections[c_id].index;
844  uint8_t other_c_id = con->connections[c_id].other_id + NUM_RESERVED_PORTS;
845  uint8_t new_data[length];
846  memcpy(new_data, data, length);
847  new_data[0] = other_c_id;
848  int ret = write_packet_TCP_secure_connection(&TCP_server->accepted_connection_array[index], new_data, length, 0);
849 
850  if (ret == -1)
851  return -1;
852 
853  return 0;
854  }
855  }
856 
857  return 0;
858 }
