294 static int create_crypto_handshake(const Net_Crypto *c, uint8_t *packet, const uint8_t *cookie, const uint8_t *nonce,
295  const uint8_t *session_pk, const uint8_t *peer_real_pk, const uint8_t *peer_dht_pubkey)
296 {
297  uint8_t plain[crypto_box_NONCEBYTES + crypto_box_PUBLICKEYBYTES + crypto_hash_sha512_BYTES + COOKIE_LENGTH];
298  memcpy(plain, nonce, crypto_box_NONCEBYTES);
299  memcpy(plain + crypto_box_NONCEBYTES, session_pk, crypto_box_PUBLICKEYBYTES);
300  crypto_hash_sha512(plain + crypto_box_NONCEBYTES + crypto_box_PUBLICKEYBYTES, cookie, COOKIE_LENGTH);
301  uint8_t cookie_plain[COOKIE_DATA_LENGTH];
302  memcpy(cookie_plain, peer_real_pk, crypto_box_PUBLICKEYBYTES);
303  memcpy(cookie_plain + crypto_box_PUBLICKEYBYTES, peer_dht_pubkey, crypto_box_PUBLICKEYBYTES);
304 
305  if (create_cookie(plain + crypto_box_NONCEBYTES + crypto_box_PUBLICKEYBYTES + crypto_hash_sha512_BYTES, cookie_plain,
306  c->secret_symmetric_key) != 0)
307  return -1;
308 
309  new_nonce(packet + 1 + COOKIE_LENGTH);
310  int len = encrypt_data(peer_real_pk, c->self_secret_key, packet + 1 + COOKIE_LENGTH, plain, sizeof(plain),
311  packet + 1 + COOKIE_LENGTH + crypto_box_NONCEBYTES);
312 
313  if (len != HANDSHAKE_PACKET_LENGTH - (1 + COOKIE_LENGTH + crypto_box_NONCEBYTES))
314  return -1;
315 
316  packet[0] = NET_PACKET_CRYPTO_HS;
317  memcpy(packet + 1, cookie, COOKIE_LENGTH);
318 
319  return HANDSHAKE_PACKET_LENGTH;
320 }

322 /* Handle a crypto handshake packet of length.
323  * put the nonce contained in the packet in nonce,
324  * the session public key in session_pk
325  * the real public key of the peer in peer_real_pk
326  * the dht public key of the peer in dht_public_key and
327  * the cookie inside the encrypted part of the packet in cookie.
328  *
329  * if expected_real_pk isn't NULL it denotes the real public key
330  * the packet should be from.
331  *
332  * nonce must be at least crypto_box_NONCEBYTES
333  * session_pk must be at least crypto_box_PUBLICKEYBYTES
334  * peer_real_pk must be at least crypto_box_PUBLICKEYBYTES
335  * cookie must be at least COOKIE_LENGTH
336  *
337  * return -1 on failure.
338  * return 0 on success.
339  */
340 static int handle_crypto_handshake(const Net_Crypto *c, uint8_t *nonce, uint8_t *session_pk, uint8_t *peer_real_pk,
341  uint8_t *dht_public_key, uint8_t *cookie, const uint8_t *packet, uint16_t length, const uint8_t *expected_real_pk)
342 {
343  if (length != HANDSHAKE_PACKET_LENGTH)
344  return -1;
345 
346  uint8_t cookie_plain[COOKIE_DATA_LENGTH];
347 
348  if (open_cookie(cookie_plain, packet + 1, c->secret_symmetric_key) != 0)
349  return -1;
350 
351  if (expected_real_pk)
352  if (public_key_cmp(cookie_plain, expected_real_pk) != 0)
353  return -1;
354 
355  uint8_t cookie_hash[crypto_hash_sha512_BYTES];
356  crypto_hash_sha512(cookie_hash, packet + 1, COOKIE_LENGTH);
357 
358  uint8_t plain[crypto_box_NONCEBYTES + crypto_box_PUBLICKEYBYTES + crypto_hash_sha512_BYTES + COOKIE_LENGTH];
359  int len = decrypt_data(cookie_plain, c->self_secret_key, packet + 1 + COOKIE_LENGTH,
360  packet + 1 + COOKIE_LENGTH + crypto_box_NONCEBYTES,
361  HANDSHAKE_PACKET_LENGTH - (1 + COOKIE_LENGTH + crypto_box_NONCEBYTES), plain);
362 
363  if (len != sizeof(plain))
364  return -1;
365 
366  if (memcmp(cookie_hash, plain + crypto_box_NONCEBYTES + crypto_box_PUBLICKEYBYTES, crypto_hash_sha512_BYTES) != 0)
367  return -1;
368 
369  memcpy(nonce, plain, crypto_box_NONCEBYTES);
370  memcpy(session_pk, plain + crypto_box_NONCEBYTES, crypto_box_PUBLICKEYBYTES);
371  memcpy(cookie, plain + crypto_box_NONCEBYTES + crypto_box_PUBLICKEYBYTES + crypto_hash_sha512_BYTES, COOKIE_LENGTH);
372  memcpy(peer_real_pk, cookie_plain, crypto_box_PUBLICKEYBYTES);
373  memcpy(dht_public_key, cookie_plain + crypto_box_PUBLICKEYBYTES, crypto_box_PUBLICKEYBYTES);
374  return 0;
375 }
