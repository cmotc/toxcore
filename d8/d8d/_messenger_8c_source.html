119  * return a uint16_t that represents the checksum of address of length len.
120  */
121 static uint16_t address_checksum(const uint8_t *address, uint32_t len)
122 {
123  uint8_t checksum[2] = {0};
124  uint16_t check;
125  uint32_t i;
126 
127  for (i = 0; i < len; ++i)
128  checksum[i % 2] ^= address[i];
129 
130  memcpy(&check, checksum, sizeof(check));
131  return check;
132 }

210 /*
211  * Add a friend.
212  * Set the data that will be sent along with friend request.
213  * Address is the address of the friend (returned by getaddress of the friend you wish to add) it must be FRIEND_ADDRESS_SIZE bytes.
214  * data is the data and length is the length.
215  *
216  * return the friend number if success.
217  * return FA_TOOLONG if message length is too long.
218  * return FAERR_NOMESSAGE if no message (message length must be >= 1 byte).
219  * return FAERR_OWNKEY if user's own key.
220  * return FAERR_ALREADYSENT if friend request already sent or already a friend.
221  * return FAERR_BADCHECKSUM if bad checksum in address.
222  * return FAERR_SETNEWNOSPAM if the friend was already there but the nospam was different.
223  * (the nospam for that friend was set to the new one).
224  * return FAERR_NOMEM if increasing the friend list size fails.
225  */
226 int32_t m_addfriend(Messenger *m, const uint8_t *address, const uint8_t *data, uint16_t length)
227 {
228  if (length > MAX_FRIEND_REQUEST_DATA_SIZE)
229  return FAERR_TOOLONG;
230 
231  uint8_t real_pk[crypto_box_PUBLICKEYBYTES];
232  id_copy(real_pk, address);
233 
234  if (!public_key_valid(real_pk))
235  return FAERR_BADCHECKSUM;
236 
237  uint16_t check, checksum = address_checksum(address, FRIEND_ADDRESS_SIZE - sizeof(checksum));
238  memcpy(&check, address + crypto_box_PUBLICKEYBYTES + sizeof(uint32_t), sizeof(check));
239 
240  if (check != checksum)
241  return FAERR_BADCHECKSUM;
242 
243  if (length < 1)
244  return FAERR_NOMESSAGE;
245 
246  if (id_equal(real_pk, m->net_crypto->self_public_key))
247  return FAERR_OWNKEY;
248 
249  int32_t friend_id = getfriend_id(m, real_pk);
250 
251  if (friend_id != -1) {
252  if (m->friendlist[friend_id].status >= FRIEND_CONFIRMED)
253  return FAERR_ALREADYSENT;
254 
255  uint32_t nospam;
256  memcpy(&nospam, address + crypto_box_PUBLICKEYBYTES, sizeof(nospam));
257 
258  if (m->friendlist[friend_id].friendrequest_nospam == nospam)
259  return FAERR_ALREADYSENT;
260 
261  m->friendlist[friend_id].friendrequest_nospam = nospam;
262  return FAERR_SETNEWNOSPAM;
263  }
264 
265  int32_t ret = init_new_friend(m, real_pk, FRIEND_ADDED);
266 
267  if (ret < 0) {
268  return ret;
269  }
270 
271  m->friendlist[ret].friendrequest_timeout = FRIENDREQUEST_TIMEOUT;
272  memcpy(m->friendlist[ret].info, data, length);
273  m->friendlist[ret].info_size = length;
274  memcpy(&(m->friendlist[ret].friendrequest_nospam), address + crypto_box_PUBLICKEYBYTES, sizeof(uint32_t));
275 
276  return ret;
277 }
