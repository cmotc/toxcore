38 static int connect_sock_to(sock_t sock, IP_Port ip_port, TCP_Proxy_Info *proxy_info)
39 {
40  if (proxy_info->proxy_type != TCP_PROXY_NONE) {
41  ip_port = proxy_info->ip_port;
42  }
43 
44  struct sockaddr_storage addr = {0};
45 
46  size_t addrsize;
47 
48  if (ip_port.ip.family == AF_INET) {
49  struct sockaddr_in *addr4 = (struct sockaddr_in *)&addr;
50 
51  addrsize = sizeof(struct sockaddr_in);
52  addr4->sin_family = AF_INET;
53  addr4->sin_addr = ip_port.ip.ip4.in_addr;
54  addr4->sin_port = ip_port.port;

56  struct sockaddr_in6 *addr6 = (struct sockaddr_in6 *)&addr;

63  return 0;
64  }
65 
66  /* nonblocking socket, connect will never return success */
67  connect(sock, (struct sockaddr *)&addr, addrsize);
68  return 1;

283 /* return 0 if pending data was sent completely
284  * return -1 if it wasn't
285  */
286 static int send_pending_data(TCP_Client_Connection *con)
287 {
288  /* finish sending current non-priority packet */
289  if (send_pending_data_nonpriority(con) == -1) {
290  return -1;
291  }
292 
293  TCP_Priority_List *p = con->priority_queue_start;
294 
295  while (p) {
296  uint16_t left = p->size - p->sent;
297  int len = send(con->sock, p->data + p->sent, left, MSG_NOSIGNAL);
298 
299  if (len != left) {
300  if (len > 0) {
301  p->sent += len;
302  }
303 
304  break;
305  }
306 
307  TCP_Priority_List *pp = p;
308  p = p->next;
309  free(pp);
310  }
311 
312  con->priority_queue_start = p;
313 
314  if (!p) {
315  con->priority_queue_end = NULL;
316  return 0;
317  }
318 
319  return -1;
320 }

361 /* return 1 on success.
362  * return 0 if could not send packet.
363  * return -1 on failure (connection must be killed).

365 static int write_packet_TCP_secure_connection(TCP_Client_Connection *con, const uint8_t *data, uint16_t length,
366  _Bool priority)
367 {
368  if (length + crypto_box_MACBYTES > MAX_PACKET_SIZE)
369  return -1;
370 
371  _Bool sendpriority = 1;
372 
373  if (send_pending_data(con) == -1) {
374  if (priority) {
375  sendpriority = 0;
376  } else {
377  return 0;
378  }
379  }
380 
381  uint8_t packet[sizeof(uint16_t) + length + crypto_box_MACBYTES];
382 
383  uint16_t c_length = htons(length + crypto_box_MACBYTES);
384  memcpy(packet, &c_length, sizeof(uint16_t));
385  int len = encrypt_data_symmetric(con->shared_key, con->sent_nonce, data, length, packet + sizeof(uint16_t));
386 
387  if ((unsigned int)len != (sizeof(packet) - sizeof(uint16_t)))
388  return -1;
389 
390  if (priority) {
391  len = sendpriority ? send(con->sock, packet, sizeof(packet), MSG_NOSIGNAL) : 0;
392 
393  if (len <= 0) {
394  len = 0;
395  }
396 
397  increment_nonce(con->sent_nonce);
398 
399  if ((unsigned int)len == sizeof(packet)) {
400  return 1;
401  }
402 
403  return add_priority(con, packet, sizeof(packet), len);
404  }
405 
406  len = send(con->sock, packet, sizeof(packet), MSG_NOSIGNAL);
407 
408  if (len <= 0)
409  return 0;
410 
411  increment_nonce(con->sent_nonce);
412 
413  if ((unsigned int)len == sizeof(packet))
414  return 1;
415 
416  memcpy(con->last_packet, packet, sizeof(packet));
417  con->last_packet_length = sizeof(packet);
418  con->last_packet_sent = len;
419  return 1;
420 }

610  */
611 TCP_Client_Connection *new_TCP_connection(IP_Port ip_port, const uint8_t *public_key, const uint8_t *self_public_key,
612  const uint8_t *self_secret_key, TCP_Proxy_Info *proxy_info)
613 {
614  if (networking_at_startup() != 0) {
615  return NULL;
616  }
617 
618  if (ip_port.ip.family != AF_INET && ip_port.ip.family != AF_INET6)
619  return NULL;
620 
621  uint8_t family = ip_port.ip.family;
622 
623  TCP_Proxy_Info default_proxyinfo;
624 
625  if (proxy_info == NULL) {
626  default_proxyinfo.proxy_type = TCP_PROXY_NONE;
627  proxy_info = &default_proxyinfo;
628  }
629 
630  if (proxy_info->proxy_type != TCP_PROXY_NONE) {
631  family = proxy_info->ip_port.ip.family;
632  }
633 
634  sock_t sock = socket(family, SOCK_STREAM, IPPROTO_TCP);
635 
636  if (!sock_valid(sock)) {
637  return NULL;
638  }
639 
640  if (!set_socket_nosigpipe(sock)) {
641  kill_sock(sock);
642  return 0;
643  }
644 
645  if (!(set_socket_nonblock(sock) && connect_sock_to(sock, ip_port, proxy_info))) {
646  kill_sock(sock);
647  return NULL;
648  }
649 
650  TCP_Client_Connection *temp = calloc(sizeof(TCP_Client_Connection), 1);
651 
652  if (temp == NULL) {
653  kill_sock(sock);
654  return NULL;
655  }
656 
657  temp->sock = sock;
658  memcpy(temp->public_key, public_key, crypto_box_PUBLICKEYBYTES);
659  memcpy(temp->self_public_key, self_public_key, crypto_box_PUBLICKEYBYTES);
660  encrypt_precompute(temp->public_key, self_secret_key, temp->shared_key);
661  temp->ip_port = ip_port;
662  temp->proxy_info = *proxy_info;
663 
664  switch (proxy_info->proxy_type) {
665  case TCP_PROXY_HTTP:

692 /* return 0 on success
693  * return -1 on failure
694  */
695 static int handle_TCP_packet(TCP_Client_Connection *conn, const uint8_t *data, uint16_t length)
696 {
697  if (length <= 1)
698  return -1;
699 
700  switch (data[0]) {
701  case TCP_PACKET_ROUTING_RESPONSE: {
702  if (length != 1 + 1 + crypto_box_PUBLICKEYBYTES)
703  return -1;
704 
705  if (data[1] < NUM_RESERVED_PORTS)
706  return 0;
707 
708  uint8_t con_id = data[1] - NUM_RESERVED_PORTS;
709 
710  if (conn->connections[con_id].status != 0)
711  return 0;
712 
713  conn->connections[con_id].status = 1;
714  conn->connections[con_id].number = ~0;
715  memcpy(conn->connections[con_id].public_key, data + 2, crypto_box_PUBLICKEYBYTES);
716 
717  if (conn->response_callback)
718  conn->response_callback(conn->response_callback_object, con_id, conn->connections[con_id].public_key);
719 
720  return 0;
721  }
722 
723  case TCP_PACKET_CONNECTION_NOTIFICATION: {
724  if (length != 1 + 1)
725  return -1;
726 
727  if (data[1] < NUM_RESERVED_PORTS)
728  return -1;
729 
730  uint8_t con_id = data[1] - NUM_RESERVED_PORTS;
731 
732  if (conn->connections[con_id].status != 1)
733  return 0;
734 
735  conn->connections[con_id].status = 2;
736 
737  if (conn->status_callback)
738  conn->status_callback(conn->status_callback_object, conn->connections[con_id].number, con_id,
739  conn->connections[con_id].status);
740 
741  return 0;
742  }
743 
744  case TCP_PACKET_DISCONNECT_NOTIFICATION: {
745  if (length != 1 + 1)
746  return -1;
747 
748  if (data[1] < NUM_RESERVED_PORTS)
749  return -1;
750 
751  uint8_t con_id = data[1] - NUM_RESERVED_PORTS;
752 
753  if (conn->connections[con_id].status == 0)
754  return 0;
755 
756  if (conn->connections[con_id].status != 2)
757  return 0;
758 
759  conn->connections[con_id].status = 1;
760 
761  if (conn->status_callback)
762  conn->status_callback(conn->status_callback_object, conn->connections[con_id].number, con_id,
763  conn->connections[con_id].status);
764 
765  return 0;
766  }
767 
768  case TCP_PACKET_PING: {
769  if (length != 1 + sizeof(uint64_t))
770  return -1;
771 
772  uint64_t ping_id;
773  memcpy(&ping_id, data + 1, sizeof(uint64_t));
774  conn->ping_response_id = ping_id;
775  send_ping_response(conn);
776  return 0;
777  }
778 
779  case TCP_PACKET_PONG: {
780  if (length != 1 + sizeof(uint64_t))
781  return -1;
782 
783  uint64_t ping_id;
784  memcpy(&ping_id, data + 1, sizeof(uint64_t));
785 
786  if (ping_id) {
787  if (ping_id == conn->ping_id) {
788  conn->ping_id = 0;
789  }
790 
791  return 0;
792  } else {
793  return -1;
794  }
795  }
796 
797  case TCP_PACKET_OOB_RECV: {
798  if (length <= 1 + crypto_box_PUBLICKEYBYTES)
799  return -1;
800 
801  if (conn->oob_data_callback)
802  conn->oob_data_callback(conn->oob_data_callback_object, data + 1, data + 1 + crypto_box_PUBLICKEYBYTES,
803  length - (1 + crypto_box_PUBLICKEYBYTES));
804 
805  return 0;
806  }
807 
808  case TCP_PACKET_ONION_RESPONSE: {
809  conn->onion_callback(conn->onion_callback_object, data + 1, length - 1);
810  return 0;
811  }
812 
813  default: {
814  if (data[0] < NUM_RESERVED_PORTS)
815  return -1;
816 
817  uint8_t con_id = data[0] - NUM_RESERVED_PORTS;
818 
819  if (conn->data_callback)
820  conn->data_callback(conn->data_callback_object, conn->connections[con_id].number, con_id, data + 1, length - 1);
821  }
822  }
823 
824  return 0;
825 }

827 static int do_confirmed_TCP(TCP_Client_Connection *conn)
828 {
829  send_pending_data(conn);
830  send_ping_response(conn);
831  send_ping_request(conn);
832 
833  uint8_t packet[MAX_PACKET_SIZE];
834  int len;
835 
836  if (is_timeout(conn->last_pinged, TCP_PING_FREQUENCY)) {
837  uint64_t ping_id = random_64b();
838 
839  if (!ping_id)
840  ++ping_id;
841 
842  conn->ping_request_id = conn->ping_id = ping_id;
843  send_ping_request(conn);
844  conn->last_pinged = unix_time();
845  }
846 
847  if (conn->ping_id && is_timeout(conn->last_pinged, TCP_PING_TIMEOUT)) {
848  conn->status = TCP_CLIENT_DISCONNECTED;
849  return 0;
850  }
851 
852  while ((len = read_packet_TCP_secure_connection(conn->sock, &conn->next_packet_length, conn->shared_key,
853  conn->recv_nonce, packet, sizeof(packet)))) {
854  if (len == -1) {
855  conn->status = TCP_CLIENT_DISCONNECTED;
856  break;
857  }
858 
859  if (handle_TCP_packet(conn, packet, len) == -1) {
860  conn->status = TCP_CLIENT_DISCONNECTED;
861  break;
862  }
863  }
864 
865  return 0;
866 }

868 /* Run the TCP connection
869  */
870 void do_TCP_connection(TCP_Client_Connection *TCP_connection)
871 {
872  unix_time_update();
873 
874  if (TCP_connection->status == TCP_CLIENT_DISCONNECTED) {
875  return;
876  }
877 
878  if (TCP_connection->status == TCP_CLIENT_PROXY_HTTP_CONNECTING) {
879  if (send_pending_data(TCP_connection) == 0) {
880  int ret = proxy_http_read_connection_response(TCP_connection);
881 
882  if (ret == -1) {
883  TCP_connection->kill_at = 0;
884  TCP_connection->status = TCP_CLIENT_DISCONNECTED;
885  }
886 
887  if (ret == 1) {
888  generate_handshake(TCP_connection);
889  TCP_connection->status = TCP_CLIENT_CONNECTING;
890  }
891  }
892  }
893 
894  if (TCP_connection->status == TCP_CLIENT_PROXY_SOCKS5_CONNECTING) {
895  if (send_pending_data(TCP_connection) == 0) {
896  int ret = socks5_read_handshake_response(TCP_connection);
897 
898  if (ret == -1) {
899  TCP_connection->kill_at = 0;
900  TCP_connection->status = TCP_CLIENT_DISCONNECTED;
901  }
902 
903  if (ret == 1) {
904  proxy_socks5_generate_connection_request(TCP_connection);
905  TCP_connection->status = TCP_CLIENT_PROXY_SOCKS5_UNCONFIRMED;
906  }
907  }
908  }
909 
910  if (TCP_connection->status == TCP_CLIENT_PROXY_SOCKS5_UNCONFIRMED) {
911  if (send_pending_data(TCP_connection) == 0) {
912  int ret = proxy_socks5_read_connection_response(TCP_connection);
913 
914  if (ret == -1) {
915  TCP_connection->kill_at = 0;
916  TCP_connection->status = TCP_CLIENT_DISCONNECTED;
917  }
918 
919  if (ret == 1) {
920  generate_handshake(TCP_connection);
921  TCP_connection->status = TCP_CLIENT_CONNECTING;
922  }
923  }
924  }
925 
926  if (TCP_connection->status == TCP_CLIENT_CONNECTING) {
927  if (send_pending_data(TCP_connection) == 0) {
928  TCP_connection->status = TCP_CLIENT_UNCONFIRMED;
929  }
930  }
931 
932  if (TCP_connection->status == TCP_CLIENT_UNCONFIRMED) {
933  uint8_t data[TCP_SERVER_HANDSHAKE_SIZE];
934  int len = read_TCP_packet(TCP_connection->sock, data, sizeof(data));
935 
936  if (sizeof(data) == len) {
937  if (handle_handshake(TCP_connection, data) == 0) {
938  TCP_connection->kill_at = ~0;
939  TCP_connection->status = TCP_CLIENT_CONFIRMED;
940  } else {
941  TCP_connection->kill_at = 0;
942  TCP_connection->status = TCP_CLIENT_DISCONNECTED;
943  }
944  }
945  }
946 
947  if (TCP_connection->status == TCP_CLIENT_CONFIRMED) {
948  do_confirmed_TCP(TCP_connection);
949  }
950 
951  if (TCP_connection->kill_at <= unix_time()) {
952  TCP_connection->status = TCP_CLIENT_DISCONNECTED;
953  }
954 }
