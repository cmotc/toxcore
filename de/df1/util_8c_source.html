116 /* state load/save */
117 int load_state(load_state_callback_func load_state_callback, void *outer,
118  const uint8_t *data, uint32_t length, uint16_t cookie_inner)
119 {
120  if (!load_state_callback || !data) {
121 #ifdef DEBUG
122  fprintf(stderr, "load_state() called with invalid args.\n");
123 #endif
124  return -1;
125  }
126 
127 
128  uint16_t type;
129  uint32_t length_sub, cookie_type;
130  uint32_t size_head = sizeof(uint32_t) * 2;
131 
132  while (length >= size_head) {
133  lendian_to_host32(&length_sub, data);
134  lendian_to_host32(&cookie_type, data + sizeof(length_sub));
135  data += size_head;
136  length -= size_head;
137 
138  if (length < length_sub) {
139  /* file truncated */
140 #ifdef DEBUG
141  fprintf(stderr, "state file too short: %u < %u\n", length, length_sub);
142 #endif
143  return -1;
144  }
145 
146  if (lendian_to_host16((cookie_type >> 16)) != cookie_inner) {
147  /* something is not matching up in a bad way, give up */
148 #ifdef DEBUG
149  fprintf(stderr, "state file garbeled: %04hx != %04hx\n", (cookie_type >> 16), cookie_inner);
150 #endif
151  return -1;
152  }
153 
154  type = lendian_to_host16(cookie_type & 0xFFFF);
155 
156  int ret = load_state_callback(outer, data, length_sub, type);
157 
158  if (ret == -1) {
159  return -1;
160  }
161 
162  /* -2 means end of save. */
163  if (ret == -2)
164  return 0;
165 
166  data += length_sub;
167  length -= length_sub;
168  }
169 
170  return length == 0 ? 0 : -1;
171 };
