84 int encrypt_data_symmetric(const uint8_t *secret_key, const uint8_t *nonce, const uint8_t *plain, uint32_t length,
85  uint8_t *encrypted)
86 {
87  if (length == 0)
88  return -1;
89 
90  uint8_t temp_plain[length + crypto_box_ZEROBYTES];
91  uint8_t temp_encrypted[length + crypto_box_MACBYTES + crypto_box_BOXZEROBYTES];
92 
93  memset(temp_plain, 0, crypto_box_ZEROBYTES);
94  memcpy(temp_plain + crypto_box_ZEROBYTES, plain, length); // Pad the message with 32 0 bytes.
95 
96  if (crypto_box_afternm(temp_encrypted, temp_plain, length + crypto_box_ZEROBYTES, nonce, secret_key) != 0)
97  return -1;
98 
99  /* Unpad the encrypted message. */
100  memcpy(encrypted, temp_encrypted + crypto_box_BOXZEROBYTES, length + crypto_box_MACBYTES);
101  return length + crypto_box_MACBYTES;
102 }

104 int decrypt_data_symmetric(const uint8_t *secret_key, const uint8_t *nonce, const uint8_t *encrypted, uint32_t length,
105  uint8_t *plain)
106 {
107  if (length <= crypto_box_BOXZEROBYTES)
108  return -1;
109 
110  uint8_t temp_plain[length + crypto_box_ZEROBYTES];
111  uint8_t temp_encrypted[length + crypto_box_BOXZEROBYTES];
112 
113  memset(temp_encrypted, 0, crypto_box_BOXZEROBYTES);
114  memcpy(temp_encrypted + crypto_box_BOXZEROBYTES, encrypted, length); // Pad the message with 16 0 bytes.
115 
116  if (crypto_box_open_afternm(temp_plain, temp_encrypted, length + crypto_box_BOXZEROBYTES, nonce, secret_key) != 0)
117  return -1;
118 
119  memcpy(plain, temp_plain + crypto_box_ZEROBYTES, length - crypto_box_MACBYTES);
120  return length - crypto_box_MACBYTES;
121 }

140 /* Increment the given nonce by 1. */
141 void increment_nonce(uint8_t *nonce)
142 {
143  uint32_t i;
144 
145  for (i = crypto_box_NONCEBYTES; i != 0; --i) {
146  ++nonce[i - 1];
147 
148  if (nonce[i - 1] != 0)
149  break;
150  }

152 /* increment the given nonce by num */
153 void increment_nonce_number(uint8_t *nonce, uint32_t num)
154 {
155  uint32_t num1, num2;
156  memcpy(&num1, nonce + (crypto_box_NONCEBYTES - sizeof(num1)), sizeof(num1));
157  num1 = ntohl(num1);
158  num2 = num + num1;
159 
160  if (num2 < num1) {
161  uint32_t i;
162 
163  for (i = crypto_box_NONCEBYTES - sizeof(num1); i != 0; --i) {
164  ++nonce[i - 1];
165 
166  if (nonce[i - 1] != 0)
167  break;
168  }
169  }
170 
171  num2 = htonl(num2);
172  memcpy(nonce + (crypto_box_NONCEBYTES - sizeof(num2)), &num2, sizeof(num2));
173 }

193 /* Create a request to peer.
194  * send_public_key and send_secret_key are the pub/secret keys of the sender.
195  * recv_public_key is public key of receiver.
196  * packet must be an array of MAX_CRYPTO_REQUEST_SIZE big.
197  * Data represents the data we send with the request with length being the length of the data.
198  * request_id is the id of the request (32 = friend request, 254 = ping request).
199  *
200  * return -1 on failure.
201  * return the length of the created packet on success.
202  */
203 int create_request(const uint8_t *send_public_key, const uint8_t *send_secret_key, uint8_t *packet,
204  const uint8_t *recv_public_key, const uint8_t *data, uint32_t length, uint8_t request_id)
205 {
206  if (MAX_CRYPTO_REQUEST_SIZE < length + 1 + crypto_box_PUBLICKEYBYTES * 2 + crypto_box_NONCEBYTES + 1 +
207  crypto_box_MACBYTES)
208  return -1;
209 
210  uint8_t nonce[crypto_box_NONCEBYTES];
211  uint8_t temp[MAX_CRYPTO_REQUEST_SIZE];
212  memcpy(temp + 1, data, length);
213  temp[0] = request_id;
214  new_nonce(nonce);
215  int len = encrypt_data(recv_public_key, send_secret_key, nonce, temp, length + 1,
216  1 + crypto_box_PUBLICKEYBYTES * 2 + crypto_box_NONCEBYTES + packet);
217 
218  if (len == -1)
219  return -1;
220 
221  packet[0] = NET_PACKET_CRYPTO;
222  memcpy(packet + 1, recv_public_key, crypto_box_PUBLICKEYBYTES);
223  memcpy(packet + 1 + crypto_box_PUBLICKEYBYTES, send_public_key, crypto_box_PUBLICKEYBYTES);
224  memcpy(packet + 1 + crypto_box_PUBLICKEYBYTES * 2, nonce, crypto_box_NONCEBYTES);
225 
226  return len + 1 + crypto_box_PUBLICKEYBYTES * 2 + crypto_box_NONCEBYTES;
227 }

229 /* Puts the senders public key in the request in public_key, the data from the request
230  * in data if a friend or ping request was sent to us and returns the length of the data.
231  * packet is the request packet and length is its length.
232  *
233  * return -1 if not valid request.
234  */
235 int handle_request(const uint8_t *self_public_key, const uint8_t *self_secret_key, uint8_t *public_key, uint8_t *data,
236  uint8_t *request_id, const uint8_t *packet, uint16_t length)
237 {
238  if (length <= crypto_box_PUBLICKEYBYTES * 2 + crypto_box_NONCEBYTES + 1 + crypto_box_MACBYTES ||
239  length > MAX_CRYPTO_REQUEST_SIZE)
240  return -1;
241 
242  if (memcmp(packet + 1, self_public_key, crypto_box_PUBLICKEYBYTES) != 0)
243  return -1;
244 
245  memcpy(public_key, packet + 1 + crypto_box_PUBLICKEYBYTES, crypto_box_PUBLICKEYBYTES);
246  uint8_t nonce[crypto_box_NONCEBYTES];
247  uint8_t temp[MAX_CRYPTO_REQUEST_SIZE];
248  memcpy(nonce, packet + 1 + crypto_box_PUBLICKEYBYTES * 2, crypto_box_NONCEBYTES);
249  int len1 = decrypt_data(public_key, self_secret_key, nonce,
250  packet + 1 + crypto_box_PUBLICKEYBYTES * 2 + crypto_box_NONCEBYTES,
251  length - (crypto_box_PUBLICKEYBYTES * 2 + crypto_box_NONCEBYTES + 1), temp);
252 
253  if (len1 == -1 || len1 == 0)
254  return -1;
255 
256  request_id[0] = temp[0];
257  --len1;
258  memcpy(data, temp + 1, len1);
259  return len1;
260 }
