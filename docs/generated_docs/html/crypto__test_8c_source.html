166  for (testno = 0; testno < 100; testno++) {

169  rand_bytes(m, mlen);

172  //Generate keypairs

174  crypto_box_keypair(pk2, sk2);

181 
182  //Encrypt all four ways

184  c2len = encrypt_data(pk1, sk2, n, m, mlen, c2);

186  c4len = encrypt_data_symmetric(k2, n, m, mlen, c4);

189  ck_assert_msg(c1len == mlen + (int)crypto_box_MACBYTES, "wrong cyphertext length");
190  ck_assert_msg(memcmp(c1, c2, c1len) == 0 && memcmp(c1, c3, c1len) == 0

193  //Decrypt all four ways

195  m2len = decrypt_data(pk1, sk2, n, c1, c1len, m2);

198 
199  ck_assert_msg(m1len == m2len && m1len == m3len && m1len == m4len, "decrypted text lengths differ");
200  ck_assert_msg(m1len == mlen, "wrong decrypted text length");
