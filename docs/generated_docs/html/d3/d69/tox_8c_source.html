124 Tox *tox_new(const struct Tox_Options *options, TOX_ERR_NEW *error)
125 {
126  if (!logger_get_global())
127  logger_set_global(logger_new(LOGGER_OUTPUT_FILE, LOGGER_LEVEL, "toxcore"));
128 
129  Messenger_Options m_options = {0};
130 
131  _Bool load_savedata_sk = 0, load_savedata_tox = 0;
132 
133  if (options == NULL) {
134  m_options.ipv6enabled = TOX_ENABLE_IPV6_DEFAULT;
135  } else {
136  if (options->savedata_type != TOX_SAVEDATA_TYPE_NONE) {
137  if (options->savedata_data == NULL || options->savedata_length == 0) {
138  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_LOAD_BAD_FORMAT);
139  return NULL;
140  }
141  }
142 
143  if (options->savedata_type == TOX_SAVEDATA_TYPE_SECRET_KEY) {
144  if (options->savedata_length != TOX_SECRET_KEY_SIZE) {
145  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_LOAD_BAD_FORMAT);
146  return NULL;
147  }
148 
149  load_savedata_sk = 1;
150  } else if (options->savedata_type == TOX_SAVEDATA_TYPE_TOX_SAVE) {
151  if (options->savedata_length < TOX_ENC_SAVE_MAGIC_LENGTH) {
152  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_LOAD_BAD_FORMAT);
153  return NULL;
154  }
155 
156  if (memcmp(options->savedata_data, TOX_ENC_SAVE_MAGIC_NUMBER, TOX_ENC_SAVE_MAGIC_LENGTH) == 0) {
157  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_LOAD_ENCRYPTED);
158  return NULL;
159  }
160 
161  load_savedata_tox = 1;
162  }
163 
164  m_options.ipv6enabled = options->ipv6_enabled;
165  m_options.udp_disabled = !options->udp_enabled;
166  m_options.port_range[0] = options->start_port;
167  m_options.port_range[1] = options->end_port;
168  m_options.tcp_server_port = options->tcp_port;
169 
170  switch (options->proxy_type) {
171  case TOX_PROXY_TYPE_HTTP:
172  m_options.proxy_info.proxy_type = TCP_PROXY_HTTP;
173  break;
174 
175  case TOX_PROXY_TYPE_SOCKS5:
176  m_options.proxy_info.proxy_type = TCP_PROXY_SOCKS5;
177  break;
178 
179  case TOX_PROXY_TYPE_NONE:
180  m_options.proxy_info.proxy_type = TCP_PROXY_NONE;
181  break;
182 
183  default:
184  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_PROXY_BAD_TYPE);
185  return NULL;
186  }
187 
188  if (m_options.proxy_info.proxy_type != TCP_PROXY_NONE) {
189  if (options->proxy_port == 0) {
190  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_PROXY_BAD_PORT);
191  return NULL;
192  }
193 
194  ip_init(&m_options.proxy_info.ip_port.ip, m_options.ipv6enabled);
195 
196  if (m_options.ipv6enabled)
197  m_options.proxy_info.ip_port.ip.family = AF_UNSPEC;
198 
199  if (!addr_resolve_or_parse_ip(options->proxy_host, &m_options.proxy_info.ip_port.ip, NULL)) {
200  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_PROXY_BAD_HOST);
201  //TODO: TOX_ERR_NEW_PROXY_NOT_FOUND if domain.
202  return NULL;
203  }
204 
205  m_options.proxy_info.ip_port.port = htons(options->proxy_port);
206  }
207  }
208 
209  unsigned int m_error;
210  Messenger *m = new_messenger(&m_options, &m_error);
211 
212  if (!new_groupchats(m)) {
213  kill_messenger(m);
214 
215  if (m_error == MESSENGER_ERROR_PORT) {
216  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_PORT_ALLOC);
217  } else if (m_error == MESSENGER_ERROR_TCP_SERVER) {
218  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_PORT_ALLOC);
219  } else {
220  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_MALLOC);
221  }
222 
223  return NULL;
224  }
225 
226  if (load_savedata_tox && messenger_load(m, options->savedata_data, options->savedata_length) == -1) {
227  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_LOAD_BAD_FORMAT);
228  } else if (load_savedata_sk) {
229  load_secret_key(m->net_crypto, options->savedata_data);
230  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_OK);
231  } else {
232  SET_ERROR_PARAMETER(error, TOX_ERR_NEW_OK);
233  }
234 
235  return m;
236 }

260 bool tox_bootstrap(Tox *tox, const char *address, uint16_t port, const uint8_t *public_key, TOX_ERR_BOOTSTRAP *error)
261 {
262  if (!address || !public_key) {
263  SET_ERROR_PARAMETER(error, TOX_ERR_BOOTSTRAP_NULL);
264  return 0;
265  }
266 
267  if (port == 0) {
268  SET_ERROR_PARAMETER(error, TOX_ERR_BOOTSTRAP_BAD_PORT);
269  return 0;
270  }
271 
272  struct addrinfo *root, *info;
273 
274  if (getaddrinfo(address, NULL, NULL, &root) != 0) {
275  SET_ERROR_PARAMETER(error, TOX_ERR_BOOTSTRAP_BAD_HOST);
276  return 0;
277  }
278 
279  info = root;
280 
281  unsigned int count = 0;
282 
283  do {
284  IP_Port ip_port;
285  ip_port.port = htons(port);
286  ip_port.ip.family = info->ai_family;
287 
288  if (info->ai_socktype && info->ai_socktype != SOCK_DGRAM) {
289  continue;
290  }
291 
292  if (info->ai_family == AF_INET) {
293  ip_port.ip.ip4.in_addr = ((struct sockaddr_in *)info->ai_addr)->sin_addr;
294  } else if (info->ai_family == AF_INET6) {
295  ip_port.ip.ip6.in6_addr = ((struct sockaddr_in6 *)info->ai_addr)->sin6_addr;
296  } else {
297  continue;
298  }
299 
300  Messenger *m = tox;
301  onion_add_bs_path_node(m->onion_c, ip_port, public_key);
302  DHT_bootstrap(m->dht, ip_port, public_key);
303  ++count;
304  } while ((info = info->ai_next));
305 
306  freeaddrinfo(root);
307 
308  if (count) {
309  SET_ERROR_PARAMETER(error, TOX_ERR_BOOTSTRAP_OK);
310  return 1;
311  } else {
312  SET_ERROR_PARAMETER(error, TOX_ERR_BOOTSTRAP_BAD_HOST);
313  return 0;
314  }
315 }
