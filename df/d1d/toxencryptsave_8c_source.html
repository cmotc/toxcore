65 {

74 /* Generates a secret symmetric key from the given passphrase. out_key must be at least
75  * TOX_PASS_KEY_LENGTH bytes long.
76  * Be sure to not compromise the key! Only keep it in memory, do not write to disk.
77  * The password is zeroed after key derivation.

93 /* Same as above, except with use the given salt for deterministic key derivation.
94  * The salt must be TOX_PASS_SALT_LENGTH bytes in length.
95  */
96 bool tox_derive_key_with_salt(const uint8_t *passphrase, size_t pplength, const uint8_t *salt, TOX_PASS_KEY *out_key,
97  TOX_ERR_KEY_DERIVATION *error)
98 {
99  if (!salt || !out_key || (!passphrase && pplength != 0)) {
100  SET_ERROR_PARAMETER(error, TOX_ERR_KEY_DERIVATION_NULL);
101  return 0;
102  }
103 
104  uint8_t passkey[crypto_hash_sha256_BYTES];
105  crypto_hash_sha256(passkey, passphrase, pplength);
106 
107  uint8_t key[crypto_box_KEYBYTES];
108 
109  /* Derive a key from the password */
110  /* http://doc.libsodium.org/key_derivation/README.html */
111  /* note that, according to the documentation, a generic pwhash interface will be created
112  * once the pwhash competition (https://password-hashing.net/) is over */
113  if (crypto_pwhash_scryptsalsa208sha256(
114  key, sizeof(key), (char *)passkey, sizeof(passkey), salt,
115  crypto_pwhash_scryptsalsa208sha256_OPSLIMIT_INTERACTIVE * 2, /* slightly stronger */
116  crypto_pwhash_scryptsalsa208sha256_MEMLIMIT_INTERACTIVE) != 0) {
117  /* out of memory most likely */
118  SET_ERROR_PARAMETER(error, TOX_ERR_KEY_DERIVATION_FAILED);
119  return 0;
120  }
121 
122  sodium_memzero(passkey, crypto_hash_sha256_BYTES); /* wipe plaintext pw */
123  memcpy(out_key->salt, salt, crypto_pwhash_scryptsalsa208sha256_SALTBYTES);
124  memcpy(out_key->key, key, crypto_box_KEYBYTES);
125  SET_ERROR_PARAMETER(error, TOX_ERR_KEY_DERIVATION_OK);
126  return 1;
127 }

129 /* Encrypt arbitrary with a key produced by tox_derive_key_*. The output
130  * array must be at least data_len + TOX_PASS_ENCRYPTION_EXTRA_LENGTH bytes long.
131  * key must be TOX_PASS_KEY_LENGTH bytes.
132  * If you already have a symmetric key from somewhere besides this module, simply
133  * call encrypt_data_symmetric in toxcore/crypto_core directly.
134  *
135  * returns true on success

137 bool tox_pass_key_encrypt(const uint8_t *data, size_t data_len, const TOX_PASS_KEY *key, uint8_t *out,
138  TOX_ERR_ENCRYPTION *error)
139 {
140  if (data_len == 0 || !data || !key || !out) {
141  SET_ERROR_PARAMETER(error, TOX_ERR_ENCRYPTION_NULL);
142  return 0;
143  }
144 
145  /* the output data consists of, in order:
146  * salt, nonce, mac, enc_data
147  * where the mac is automatically prepended by the encrypt()
148  * the salt+nonce is called the prefix
149  * I'm not sure what else I'm supposed to do with the salt and nonce, since we
150  * need them to decrypt the data
151  */
152 
153  /* first add the magic number */
154  memcpy(out, TOX_ENC_SAVE_MAGIC_NUMBER, TOX_ENC_SAVE_MAGIC_LENGTH);
155  out += TOX_ENC_SAVE_MAGIC_LENGTH;
156 
157  /* then add the rest prefix */
158  memcpy(out, key->salt, crypto_pwhash_scryptsalsa208sha256_SALTBYTES);
159  out += crypto_pwhash_scryptsalsa208sha256_SALTBYTES;
160 
161  uint8_t nonce[crypto_box_NONCEBYTES];
162  random_nonce(nonce);
163  memcpy(out, nonce, crypto_box_NONCEBYTES);
164  out += crypto_box_NONCEBYTES;
165 
166  /* now encrypt */
167  if (encrypt_data_symmetric(key->key, nonce, data, data_len, out)
168  != data_len + crypto_box_MACBYTES) {
169  SET_ERROR_PARAMETER(error, TOX_ERR_ENCRYPTION_FAILED);
170  return 0;
171  }
172 
173  SET_ERROR_PARAMETER(error, TOX_ERR_ENCRYPTION_OK);
174  return 1;
175 }
